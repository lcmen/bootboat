Modal

Dialog overlay. Two patterns: body-only, or with footer actions. Requires Alpine.js.

Two patterns

<!-- Trigger -->
<button @click="showModal = true" class="btn btn-primary">Open</button>

<!-- Modal (teleported to body) -->
<template x-teleport="body">
  <div x-show="showModal" x-cloak>
    <div class="modal-overlay" x-transition.opacity @click="showModal = false"></div>
      <div class="modal-wrapper" x-transition>
      <!-- Add .modal-lg (max-w-lg) or .modal-xl (max-w-xl) for larger sizes -->
      <div class="modal" @click.outside="showModal = false">
        <div class="modal-body">
          <button @click="showModal = false" class="modal-close">
            <svg class="h-6 w-6">...</svg>
          </button>
          <h3 class="h3">Title</h3>
          <p class="mt-2 text-body">Content</p>
        </div>
        <!-- Optional footer: -->
        <div class="modal-footer">
          <button @click="showModal = false" class="btn btn-secondary">Cancel</button>
          <button @click="showModal = false" class="btn btn-danger">Confirm</button>
        </div>
      </div>
    </div>
  </div>
</template>
