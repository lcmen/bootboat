Dropdown

Requires Alpine.js. Accessible with aria-expanded and Escape to close.

Default dropdown with sections

Settings Help
Log out 
<div
  x-data="{ open: false, toggle() { ... }, close(focusAfter) { ... } }"
  x-on:keydown.escape.prevent.stop="close($refs.button)"
  x-on:focusin.window="! $refs.panel.contains($event.target) && close()"
  x-id="['dd']"
  class="dropdown">

  <button x-ref="button" x-on:click="toggle()"
    :aria-expanded="open" :aria-controls="$id('dd')"
    class="dropdown-trigger">
    Options <svg ...></svg>
  </button>

  <div x-ref="panel" x-show="open" x-transition.origin.top.left
    x-on:click.outside="close($refs.button)"
    :id="$id('dd')"
    class="dropdown-panel dropdown-panel-left">
    <div class="dropdown-section">
      <a class="dropdown-item">
        <svg class="dropdown-item-icon">...</svg> Settings
      </a>
    </div>
    <div class="dropdown-section">
      <a class="dropdown-item-danger">Log out</a>
    </div>
  </div>
</div>
